# To create a new React project:

    npm create vite@latest

# Install npm modules:
    npm install
    npm install nes.css
    npm install react-cookie
    npm install -D sass
    npm install jwt-decode
    npm install react-router-dom
    npm install xp.css
    npm i react-draggable

    <!-- npm i typescript-cookie -->

# Run server:

    npm run dev
